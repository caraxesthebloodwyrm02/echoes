```python
#!/usr/bin/env python3
"""
Fused Terminal Chat Assistant

- Advanced CLI + ChatBot style, combining intelligent communication,
  experienced terminal skills, and practical user focus.
"""

# --- Imports: (merge, see real code for all details) ---
import os, sys, time, json, logging, argparse, threading, re
from terminal_assistant import TerminalAssistant # (migrated code)
# ... import for TTS, speech, chatbot if wanted
# ... (add Chat + voice support if requested, from assistant.py)

# --- Logging/config ---
logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')
logger = logging.getLogger("FusedAssistant")

# --- Fused main loop ---
def main():
    parser = argparse.ArgumentParser(description="Fused Terminal Chat & Command Assistant")
    parser.add_argument("--chat", action="store_true", help="Chat mode (AI assistant conversation)")    
    parser.add_argument("--cmd", help="Command mode - run a one-liner terminal operation")
    parser.add_argument("--session", help="Load previous session file")

    args = parser.parse_args()
    assistant = TerminalAssistant()  # All battle-tested tools inside

    # Load session if needed
    if args.session:
        assistant.load_session(args.session)

    try:
        if args.chat:
            print("ü§ñ Welcome to Chat Mode!")
            while True:
                user_input = input("You: ").strip()
                if user_input.lower() in ['quit', 'exit']:
                    print("üëã Bye!")
                    break
                response = assistant.execute_command(user_input)
                print(f"Assistant: {response.output or response.error}")
        elif args.cmd:
            # Single command line
            res = assistant.execute_command(args.cmd)
            print(res.output if res.success else f"Error: {res.error}")
        else:
            # Menu mode: ask user which mode, default to chat
            print("Select mode: [1] Chat (default)  [2] Command")
            mode_sel = input("Mode: ").strip()
            if mode_sel == "2":
                user_cmd = input("Enter command: ").strip()
                res = assistant.execute_command(user_cmd)
                print(res.output if res.success else f"Error: {res.error}")
            else:
                print("ü§ñ Chat Mode started. Type 'quit' to exit.")
                while True:
                    user_input = input("You: ").strip()
                    if user_input.lower() in ['quit', 'exit']:
                        print("üëã Bye!")
                        break
                    response = assistant.execute_command(user_input)
                    print(f"Assistant: {response.output or response.error}")
    except KeyboardInterrupt:
        print("\nüëã Interrupted. Exiting...")

if __name__ == "__main__":
    main()
```
